.text
.global _start

_start:
        mov     r1, #1 //Le asignamos el valor de 1 al registro 1
        mov     r2, #2 //Le asignamos el valor de 2 al registro 2
        cmp     r1, r2 //Comparación entre el registro 1 y 2
        bne     else   //Llamada a la subrutina else

if:
        add     r0, r1, #5 //Sumar 5 al registro 1 y guardar en registro 0
        b       salir //Llamada a la subrutina de salida

else:
        add     r0, r1, #6 //Sumar 6 al registro 1 y guardar en registro 0
salir:
        mov     r7, #1 //Sentencia de salida
        svc     0
